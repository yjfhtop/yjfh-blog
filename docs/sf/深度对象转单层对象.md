# 深度对象转单层对象

>   { a: { b: [], c: 10 } } 转为 { a.b: [], a.c: 10 }

### 思路 
递归就完事了...甚至都不需要注释

### 实现代码
```js
function getDeepKeyObj(obj) {
    const targetObj = {}
    function useFun(obj, pKeyArr = []) {
        Object.keys(obj).forEach(key => {
            const item = obj[key]
            const keyArr = [...pKeyArr, key]
            if (({}).toString.call(item) === '[object Object]') {
                useFun(item, keyArr)
            } else {

                targetObj[keyArr.join('.')] = item
            }
        })
    }
    useFun(obj, [])
    return targetObj

}
```
